```bash
#!/bin/bash
################ Example Shell Script #################################

# List directory contents
ls
# Print working directory
pwd
# Change directory
cd /path/to/directory
# Open Vim text editor (create or edit a file)
vi filename
# Open Nano text editor (create or edit a file)
nano filename
# Move or rename a file
mv source destination
# Remove a file
rm filename
# Remove a directory (recursively)
rm -r directory
# Remove a directory (recursively force)
rm -rf directory 
# Create a new directory
mkdir new_directory
# Create an empty file or update timestamp
touch filename
# Ping a host to check network connectivity
ping example.com
```
